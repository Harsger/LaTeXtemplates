\frame{\frametitle{Summary}
	\footnotesize
	\begin{itemize}
		\item
			Upgrade of the ATLAS Muon Spectrometer Inner End Caps
			
			\begin{tabular}{lcl}			
				sTGC and Micromegas & $\Rightarrow$ & triggering and precision tracking
				\\
				Micromegas & $\Rightarrow$ & German participation in construction
			\end{tabular}
			\vspace{3mm}
		\item
			\textbf{Construction} of \SI{2}{\square\meter}-sized electrode panels
			
			\begin{tabular}{lcl}
				narrow amplification gap & $\Rightarrow$ & superior planarity required
				\\
				threefold anode & $\Rightarrow$ & precise alignment required
			\end{tabular}
			\vspace{3mm}
		\item
			\textbf{Commissioning} of four-layerd Micromegas Modules
					
			\begin{tabular}{lcl}	
				wet cleaning & $\Rightarrow$ & removal of production remnants
				\\
				dry cleaning & $\Rightarrow$ & removal of immediate dust
			\end{tabular}
			\vspace{3mm}
		\item
			\textbf{Validation} using cosmic muons
				
			\begin{tabular}{lcl}						
				high efficiency & $\Rightarrow$ & function of amplification
				\\
				good resolution & $\Rightarrow$ & alignment reconstruction
			\end{tabular}
			\vspace{3mm}
	\end{itemize}
