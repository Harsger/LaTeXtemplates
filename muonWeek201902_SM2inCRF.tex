\frame{\frametitle{Summary}
	\footnotesize
	\begin{itemize}
		\item
			noise of preliminary APV electronics limits efficiency
			
			$\Rightarrow$ improve grounding for next measurements
			
			\vspace{4mm}
		\item
			no concluding comparison of alignment measurements
			
			$\Rightarrow$ more statics will improve understanding of systematics
						
			\vspace{4mm}
		\item
			very good behavior with different gas mixture of Ar:CO$_{2}$ 80:20 vol\%
			
			$\Rightarrow$ resolution (position and time) has to be further investigated
	\end{itemize}
