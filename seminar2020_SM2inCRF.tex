\frame{\frametitle{Summary}
	\scriptsize
	\begin{itemize}
		\item
			Upgrade of the ATLAS Muon Spectrometer Inner End Caps
			
			\begin{tabular}{lcl}			
				sTGC and Micromegas & $\Rightarrow$ & triggering and precision tracking
				\\
				Micromegas & $\Rightarrow$ & German participation in construction
			\end{tabular}
%			\vspace{3mm}
		\item
			\textbf{Construction} of \SI{2}{\square\meter}-sized electrode panels
			
			\begin{tabular}{lcl}
				narrow amplification gap ($\sim$ 0.12 mm) & $\Rightarrow$ & superior planarity required ($\sim$ 0.02 mm)
				\\
				threefold anode & $\Rightarrow$ & precise alignment required ($\sim$ 0.05 mm)
			\end{tabular}
%			\vspace{3mm}
		\item
			\textbf{Commissioning} of four-layerd Micromegas Modules
					
			\begin{tabular}{lcl}	
				wet cleaning & $\Rightarrow$ & removal of production remnants
				\\
				dry cleaning & $\Rightarrow$ & removal of immediate dust
				\\
				active area preparation & $\Rightarrow$ & HV-stability ($<$ 20 nA @ 580 V)
			\end{tabular}
%			\vspace{3mm}
		\item
			\textbf{Validation} using cosmic muons
				
			\begin{tabular}{lcl}						
				high efficiency ($>$ 90 \%) & $\Rightarrow$ & function of amplification
				\\
				good resolution ($<$ 0.2 mm) & $\Rightarrow$ & alignment reconstruction
			\end{tabular}
%			\vspace{3mm}
	\end{itemize}
